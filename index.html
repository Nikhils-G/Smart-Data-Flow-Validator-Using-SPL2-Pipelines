// SPL2 pipeline for comprehensive data health monitoring // Detects anomalies, calculates health metrics, and visualizes results from splunk_logs where timestamp >= relative_time(now(), "-4h") | timechart span=1m count as event_count, count(eval(log_level="ERROR")) as errors, count(eval(log_level="WARN")) as warnings by source | fillnull value=0 event_count, errors, warnings // Calculate expected baseline using moving average | eval expected_count = avg(event_count) over all_time, error_rate = errors / event_count, warning_rate = warnings / event_count // Detect missing logs (gaps in sequence) | streamstats current=f window=2 values(_time) as time_window | eval time_diff = time_window[1] - time_window[0], is_gap = if(time_diff > 60, 1, 0) // 1 minute threshold // Detect spikes (200% increase from baseline) | eval spike_threshold = expected_count * 2, is_spike = if(event_count > spike_threshold, 1, 0), spike_severity = if(is_spike=1, (event_count - expected_count)/expected_count, 0) // Calculate advanced health metrics | eventstats sum(is_gap) as total_gaps, sum(is_spike) as total_spikes, stdev(event_count) as event_stdev, avg(event_count) as event_avg | eval gap_penalty = (total_gaps / count()) * 100, spike_penalty = (total_spikes / count()) * 50, consistency_score = 100 - (event_stdev / event_avg) * 100, error_penalty = sum(error_rate) * 10, warning_penalty = sum(warning_rate) * 5 // Composite health score (0-100) with weighted factors | eval health_score = 100 - (gap_penalty * 0.3) - (spike_penalty * 0.2) - (error_penalty * 0.3) - (warning_penalty * 0.2) + (consistency_score * 0.5) | stats max(health_score) as health_score, sum(is_gap) as missing_logs, sum(is_spike) as anomaly_spikes, avg(error_rate) as avg_error_rate, avg(warning_rate) as avg_warning_rate | eval health_status = case( health_score >= 80, "OPTIMAL", health_score >= 60, "STABLE", health_score >= 40, "DEGRADED", true, "CRITICAL" )
